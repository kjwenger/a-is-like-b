    val := reflect.ValueOf(slice)
    out := reflect.MakeSlice(reflect.TypeOf(slice), val.Len(), val.Cap())
    for i := 0; i < val.Len(); i++ {

            reflect.ValueOf(fn(val.Index(i).Interface())),

class TailEndRecursion {
    @TailRecursive
    static def factorial(n, a = 1) {
        if (n < 0) return 0
        else if (n == 0) return 1
